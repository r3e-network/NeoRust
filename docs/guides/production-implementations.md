pub fn script_hash_from_string(s: &str) -> String {
    // Use SHA256 for proper hashing
    let mut hasher = Sha256::new();
    hasher.update(s.as_bytes());
    let result = hasher.finalize();
    format!("0x{}", hex::encode(result))
}
